255void cppMD01Frame::PopulateMasterGrid()
256{
257 // Attempt to connect to the MySQL database
258 if (!conn.connect("bcswebtools", "localhost", "bcs", "Peace007!amb"))
259 {
260 // Database connection failed, exit the function
261 // wxWrite("Database connection failed.");
262 return;
263 }
264
265 // Execute the SQL query to fetch all records from the 'lk1' table, ordered by 'cat'
266 mysqlpp::Query query = conn.query("SELECT * FROM lk1 ORDER BY cat");
267 mysqlpp::StoreQueryResult res = query.store();
268
269 // Clear the current contents of the grid and append rows based on the number of records fetched
270 masterGrid->ClearGrid();
271 masterGrid->AppendRows(res.num_rows());
272
273 // Populate the grid with data from the result set
274 for (size_t i = 0; i < res.num_rows(); ++i)
275 {
276 masterGrid->SetCellValue(i, 0, res[i][0].c_str()); // Set cell value for column 0
277 masterGrid->SetCellValue(i, 1, res[i][1].c_str()); // Set cell value for column 1
278 masterGrid->SetCellValue(i, 2, res[i][2].c_str()); // Set cell value for column 2
279 }

290void cppMD01Frame::OnCellClick(wxGridEvent& event)
291{
292 // Get the grid from the event
293 wxGrid* grid = static_cast<wxGrid*>(event.GetEventObject());
294
295 // Get the clicked cell's row and column
296 row = event.GetRow();
297 int col = event.GetCol();
298
299 // Retrieve the cell value as a wxString
300 wxString cellValue = grid->GetCellValue(row, col);
301
302 // Check if the cell value starts with "http"
303 if (cellValue.StartsWith("http"))
304 {
305 // Open the URL in the default web browser
306 wxLaunchDefaultBrowser(cellValue);
307 }
308 else if (cellValue.find(".pdf") != wxNOT_FOUND)
309 {
310 cmd = "okular \"";
311 std::string tbuf;
312 tbuf = cellValue.ToStdString();
313 cmd = cmd + tbuf + "\"";
314 system(cmd.c_str());
315 }
316 else if (cellValue.find(".odt") != wxNOT_FOUND)
317 {
318 std::string pbuf = "\"" + cellValue.ToStdString() + "\" & ";
319 cmd = "openoffice4 -writer " + pbuf;
320 system(cmd.c_str());
321 }
322 else
323 {
324 system(cellValue.c_str());
325 }
326 // Allow the event to continue to be processed
327 event.Skip();
328}

510void cppMD01Frame::ListFiles(const wxString& dir, const wxString& filespec)
511{
512 wxArrayString *fileList;
513 fileList->Clear();
514
515 fs::path path = std::string(dir.mb_str());
516 std::string spec = std::string(filespec.mb_str());
517
518 if (fs::exists(path) && fs::is_directory(path))
519 {
520 for (const auto& entry : fs::directory_iterator(path))
521 {
522 if (fs::is_regular_file(entry))
523 {
524 std::string filename = entry.path().filename().string();
525
526 // Apply filespec filtering
527 if (boost::iends_with(filename, spec.substr(1))) // "*.ext" -> "ext"
528 {
529 fileList->Add(wxString::FromUTF8(filename));
530 }
531 }
532 }
533 }
